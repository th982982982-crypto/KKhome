  function renderDynamicTable() {
    var config = UI_CONFIG[currentTab]; if (!config) return;
    document.getElementById("mainTaxTable") ? document.getElementById("mainTaxTable").className = "table-custom pinned-mode" : null;
    if (!Array.isArray(globalData) || globalData.length === 0) {
      document.getElementById("tableContainer").innerHTML = '<div style="text-align:center; padding:50px; color:#999;"><p>📭 Cơ sở dữ liệu trống - Vui lòng chọn và nạp file XML tờ khai thuế!</p></div>'; return;
    }

    var filteredData = globalData.filter(function(d) {
      if (d.type !== currentTab) return false;
      if (selectedMst !== 'all' && d.mst.toString().trim() !== selectedMst) return false;
      if (d.status !== "ĐƯỢC CỘNG") return false;
      return true;
    });

    var html = '<table id="mainTaxTable" class="table-custom pinned-mode"><thead><tr><th class="text-center" style="font-weight: bold; text-align: center;">CHỈ TIÊU BÁO CÁO (TỜ KHAI CHUẨN)</th><th class="text-center" style="font-weight: bold; text-align: center;">MÃ SỐ</th>';

    if (currentMode === 'year') {
      var colsYears = filteredData.map(function(d) { return d.year.toString().trim(); }).filter(function(y) { return y !== ""; });
      var uniqueYearsForColumns = colsYears.filter(function(item, pos) { return colsYears.indexOf(item) == pos; }).sort(function(a, b) { return a - b; });
      var displayYears = (selectedYear === 'all') ? (uniqueYearsForColumns.length > 0 ? uniqueYearsForColumns : [new Date().getFullYear().toString()]) : [selectedYear];

      displayYears.forEach(function(yr) { html += '<th class="text-center" style="min-width:150px; font-weight: bold; text-align: center;">TỔNG NĂM ' + yr + '</th>'; });
      html += '</tr></thead><tbody>';

      config.forEach(function(row) {
        if (row.isHeader) {
          html += '<tr class="section-header-row"><td colspan="' + (displayYears.length + 2) + '"><div style="position: sticky !important; left: 15px !important; padding: 12px 5px; font-weight: bold; color: #c47c1a; font-size: 13px; text-transform: uppercase; width: max-content;">' + row.name + '</div></td></tr>';
          return;
        }
        html += '<tr><td>' + row.name + '</td><td class="text-center" style="color:#5c7080;">[' + row.code + ']</td>';
        displayYears.forEach(function(yr) {
          var sum = filteredData.filter(function(d) { return d.code.toString().trim() === row.code.toString().trim() && d.year === yr; }).reduce(function(t, i) { return t + i.value; }, 0);
          html += '<td class="text-right" style="' + (sum === 0 ? 'color: #004085;' : 'color: #001f3f;') + '">' + formatNumber(sum) + '</td>';
        });
        html += '</tr>';
      });
    } else {
      var rawPeriods = filteredData.filter(function(d) { return selectedYear === 'all' || d.year === selectedYear; }).map(function(d) { return d.period; });
      var periods = rawPeriods.filter(function(item, pos, self) { return self.indexOf(item) == pos; });

      periods.sort(function(a, b) {
        var pA = a.split('/'); var pB = b.split('/');
        if (pA.length < 2 || pB.length < 2) return a.localeCompare(b);
        var yA = parseInt(pA[1]), yB = parseInt(pB[1]); var mA = parseInt(pA[0]), mB = parseInt(pB[0]);
        if (yA !== yB) return yA - yB; return mA - mB;
      });

      if (periods.length === 0) {
        html += '<th class="text-right">KỲ CHƯA CÓ DỮ LIỆU</th></tr></thead><tbody><tr><td colspan="3" style="text-align:center; padding:30px; color:#999;">Không tìm thấy kỳ kê khai nào!</td></tr>';
      } else {
        periods.forEach(function(prd) {
          var sample = filteredData.find(function(d) { return d.period === prd; }) || {};
          var lk = sample.loaiKhai === 'C' ? 'Chính thức' : (sample.loaiKhai === 'B' ? 'Bổ sung' : sample.loaiKhai || 'C');
          var sl = sample.soLan !== undefined ? 'Lần ' + sample.soLan : 'Lần 0'; var st = sample.status || 'ĐƯỢC CỘNG';
          html += '<th class="text-center" style="min-width:160px; font-size:12.5px; font-weight: bold; text-align: center; vertical-align: top;">' +
                    'KỲ ' + prd + '<br><span style="font-size:10.5px; font-weight:bold; color:#c47c1a;">[' + lk + ' - ' + sl + ']</span><br><span style="font-size:9.5px; font-weight:bold; color:#28a745;">● ' + st + '</span></th>';
        });
        html += '</tr></thead><tbody>';

        config.forEach(function(row) {
          if (row.isHeader) {
            html += '<tr class="section-header-row"><td colspan="' + (periods.length + 2) + '"><div style="position: sticky !important; left: 15px !important; padding: 12px 5px; font-weight: bold; color: #c47c1a; font-size: 13px; text-transform: uppercase; width: max-content;">' + row.name + '</div></td></tr>';
            return;
          }
          html += '<tr><td>' + row.name + '</td><td class="text-center" style="color:#5c7080;">[' + row.code + ']</td>';
          periods.forEach(function(prd) {
            var val = filteredData.find(function(d) { return d.code.toString().trim() === row.code.toString().trim() && d.period === prd; });
            var sum = val ? val.value : 0;
            html += '<td class="text-right" style="' + (sum === 0 ? 'color: #004085;' : 'color: #001f3f;') + '">' + formatNumber(sum) + '</td>';
          });
          html += '</tr>';
        });
      }
    }
    html += '</tbody></table>';
    document.getElementById("tableContainer").innerHTML = html;
  }

  async function handleFileSelect(input) {
    var files = input.files; if (!files || files.length === 0) return;
    var total = files.length; var ok = 0; var fail = 0; var tenantToken = localStorage.getItem("tenantUser");
    for (var i = 0; i < total; i++) {
      var file = files[i]; document.getElementById("uploadSubStatus").innerText = "⏳ Đang xử lý (" + (i + 1) + "/" + total + "): " + file.name;
      try {
        var text = await new Promise(function(r) { var rd = new FileReader(); rd.onload = function(e) { r(e.target.result); }; rd.readAsText(file); });
        var xmlDoc = new DOMParser().parseFromString(text, "text/xml");
        var mst = xmlDoc.querySelector("mst") ? xmlDoc.querySelector("mst").textContent : "";
        var ky = xmlDoc.querySelector("kyKKhai") ? xmlDoc.querySelector("kyKKhai").textContent : "";
        var ma = xmlDoc.querySelector("maTKhai") ? xmlDoc.querySelector("maTKhai").textContent : "";
        var nk = xmlDoc.querySelector("nguoiKy") ? xmlDoc.querySelector("nguoiKy").textContent : "Hệ thống";
        if (!mst || !ky || !ma) { fail++; continue; }
        var type = ma === "842" ? "GTGT" : (ma === "892" ? "TNDN" : "TNCN"); var yr = ky.includes("/") ? ky.split("/")[1] : ky;
        var base64 = await new Promise(function(r) { var rd = new FileReader(); rd.onload = function(e) { r(e.target.result); }; rd.readAsDataURL(file); });
        await runServerFunction("uploadAndLogFileAndParse", base64, file.name, mst, yr, type, ky, nk, tenantToken); ok++;
      } catch (err) { fail++; }
    }
    alert("🎉 Xử lý SaaS hoàn tất!\n- Thành công: " + ok + "/" + total + "\n- Lỗi: " + fail);
    document.getElementById("uploadSubStatus").innerText = "Kéo thả các tệp XML hoặc nhấn để chọn";
    input.value = "";
    loadDataFromServer();
  }
